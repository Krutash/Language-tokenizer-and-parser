<program> -> PROGRAM ROOP ROCL BROP <stmtlist> SEMICOLON BRCL
<stmtlist> -> <stmt> SEMICOLON <stmtlist> SEMICOLON | <stmt> SEMICOLON
<stmt> -> <declarationlist> | <assignmentlist>
<declarationlist> -> <declaration> SEMICOLON <declarationlist> | <declaration>
<assignmentlist> -> <assignemnt> SEMICOLON <assignemntlist> | <assignment>
<declaration> -> DECLARE <declaretype> COLON <typespecifier>
<declaretype> -> LIST OF VARIABLES STRINGCONST | STRINGCONST
<typespecifier> -> INTEGER | REAL | BOOLEAN | <arrays>
<arrays> -> <reactangular> | <jaggedarray>
<rectangular> -> ARRAY <rangedim> OF INTEGER
<rangedim> -> SQOP <low> DOT DOT <high> SQCL <rangedim> | SQOP <low> DOT DOT <high> SQCL
<jaggadarray> -> JAGGED ARRAY SQOP <low> DOT DOT <high> SQCL <bracketlist> OF INTEGER SEMICOLON <jaggedlist>
<bracketlist> -> SQOP SQCL <bracketlist> | SQOP SQCL
<jaggedlist> -> <R> SQOP NUM SQCL COLON SIZE NUM COLON VALUES BROP <arglist> BRCL
<R> -> R1| R2| R3
<arglist> -> <args> SEMICOLON <arglist> | <args>
<args> -> NUMLIST
<assignment> -> <stringconst> EQUAL <exp>
<exp> -> <arithmatic> | <logical>
<arithmatic> -> <term> PLUS <term> | <term> MINUS <term> | <term>
<term> -> <term> MUL <factor> | <term> DIV <factor> | <factor>
<factor> -> STRINGCONST
<logical> -> <factor> AND <logical> | <factor> OR <logical> | <factor>
