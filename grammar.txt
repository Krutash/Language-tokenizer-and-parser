<program> -> PROGRAM ROOP ROCL BROP <stmtlist> SEMICOLON BRCL
<stmtlist> -> <stmt> SEMICOLON <stmtlist>
<stmtlist> -> EPS
<stmt> -> <declarationlist>
<stmt> -> <assignmentlist>
<declarationlist> -> <declaration> SEMICOLON <declarationlist>
<declarationlist> -> EPS
<assignmentlist> -> <assignemnt> SEMICOLON <assignemntlist>
<assignmentlist> -> EPS
<declaration> -> DECLARE <declaretype> COLON <typespecifier>
<declaretype> -> LIST OF VARIABLES STRINGCONST
<declaretype> -> STRINGCONST
<typespecifier> -> INTEGER
<typespecifier> -> REAL
<typespecifier> -> BOOLEAN
<typespecifier> -> <arrays>
<arrays> -> <reactangular>
<arrays> -> <jaggedarray>
<rectangular> -> ARRAY <rangedim> OF INTEGER
<rangedim> -> SQOP <low> DOT <high> SQCL <rangedim>
<rangedim> -> EPS
<jaggadarray> -> JAGGED ARRAY SQOP <low> DOT <high> SQCL <bracketlist> OF INTEGER SEMICOLON <jaggedlist>
<low> -> NUM
<low> -> STRINGCONST
<high> -> NUM
<high> -> STRINGCONST
<bracketlist> -> SQOP SQCL <bracketlist>
<bracketlist> -> EPS
<jaggedlist> -> <R> SQOP NUM SQCL COLON SIZE NUM COLON VALUES BROP <arglist> BRCL
<R> -> RONE
<R> -> RTWO
<R> -> RTHREE
<arglist> -> <args> SEMICOLON <arglist>
<arglist> -> EPS
<args> -> NUMLIST
<assignment> -> <stringconst> EQUAL <exp>
<exp> -> <arithmatic>
<exp> -> <logical>
<arithmatic> -> <term> PLUS <term>
<arithmatic> -> <term> MINUS <term>
<arithmatic> -> <term>
<term> -> <term> MUL <factor>
<term> -> <term> DIV <factor>
<term> -> <factor>
<logical> -> <factor> AND <logical>
<logical> -> <factor> OR <logical>
<logical> -> <factor>
<factor> -> STRINGCONST
